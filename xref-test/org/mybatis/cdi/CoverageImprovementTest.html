102   void serializableMapperProxyShouldUnwrapInvocationTargetException() throws Throwable {
103     MyBatisBean bean = new MyBatisBean("id", (Class<Type>) (Type) ThrowingMapper.class, Set.of(), null);
104     CreationalContext<Object> creationalContext = mock(CreationalContext.class);
105     SqlSessionFactory sessionFactory = mock(SqlSessionFactory.class);
106     SqlSessionManagerRegistry registry = mock(SqlSessionManagerRegistry.class);
107     SqlSessionManager manager = mock(SqlSessionManager.class);
108 
109     when(registry.getManager(sessionFactory)).thenReturn(manager);
110     when(manager.getMapper(ThrowingMapper.class)).thenReturn(new ThrowingMapperImpl());
111 
112     try (MockedStatic<CDIUtils> cdiUtils = mockStatic(CDIUtils.class)) {
113       cdiUtils.when(() -> CDIUtils.findSqlSessionFactory(null, bean.getQualifiers(), creationalContext))
114           .thenReturn(sessionFactory);
115       cdiUtils.when(() -> CDIUtils.getRegistry(creationalContext)).thenReturn(registry);
116 
117       SerializableMapperProxy<Object> proxy = new SerializableMapperProxy<>(bean, creationalContext);
118       Method method = ThrowingMapper.class.getMethod("boom");
119 
120       Throwable thrown = assertThrows(IllegalStateException.class, () -> proxy.invoke(null, method, null));
121       assertEquals("boom", thrown.getMessage());
122     }
123   }

217   @SuppressWarnings("unchecked")
218   @Test
219   void beanKeyShouldSupportComparisonEqualityAndStringRepresentation() throws Exception {
220     Class<?> beanKeyClass = Class.forName("org.mybatis.cdi.MybatisExtension$BeanKey");
221     Constructor<?> constructor = beanKeyClass.getDeclaredConstructor(Class.class, Set.class);
222     constructor.setAccessible(true);
223 
224     Set<Annotation> manager1Annotations = new HashSet<>(
225         Arrays.asList(FooService.class.getDeclaredField("userMapper").getAnnotations()));
226     Set<Annotation> manager2Annotations = new HashSet<>(
227         Arrays.asList(FooService.class.getDeclaredField("userMapper2").getAnnotations()));
228 
229     Object manager1Key = constructor.newInstance((Class<Type>) (Type) UserMapper.class, manager1Annotations);
230     Object manager1KeyCopy = constructor.newInstance((Class<Type>) (Type) UserMapper.class, manager1Annotations);
231     Object manager2Key = constructor.newInstance((Class<Type>) (Type) UserMapper.class, manager2Annotations);
232 
233     Method compareTo = beanKeyClass.getDeclaredMethod("compareTo", beanKeyClass);
234     compareTo.setAccessible(true);
235     Method getKey = beanKeyClass.getDeclaredMethod("getKey");
236     getKey.setAccessible(true);
237 
238     assertEquals(0, compareTo.invoke(manager1Key, manager1KeyCopy));
239     assertNotEquals(0, compareTo.invoke(manager1Key, manager2Key));
240     assertEquals(manager1Key, manager1KeyCopy);
241     assertNotEquals(manager1Key, manager2Key);
242     assertNotEquals(manager1Key, null);
243     assertNotEquals(manager1Key, "other");
244     assertEquals(manager1Key.hashCode(), manager1KeyCopy.hashCode());
245     assertEquals(getKey.invoke(manager1Key), manager1Key.toString());
246   }
