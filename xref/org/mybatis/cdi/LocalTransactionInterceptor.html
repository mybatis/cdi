16  package org.mybatis.cdi;
17  
18  import java.io.Serializable;
19  import java.lang.reflect.InvocationTargetException;
20  import java.lang.reflect.UndeclaredThrowableException;
21  
22  import javax.inject.Inject;
23  import javax.interceptor.AroundInvoke;
24  import javax.interceptor.Interceptor;
25  import javax.interceptor.InvocationContext;
26  import javax.transaction.HeuristicMixedException;
27  import javax.transaction.HeuristicRollbackException;
28  import javax.transaction.NotSupportedException;
29  import javax.transaction.RollbackException;
30  import javax.transaction.SystemException;
31  
32  import org.apache.ibatis.session.SqlSessionManager;
33  
34  /**
35   * Best-effort interceptor for local transactions. It locates all the instances
36   * of {@code SqlSssionManager} and starts transactions on all them. It cannot
37   * guarantee atomiticy if there is more than one {@code SqlSssionManager}. Use
38   * XA drivers, a JTA container and the {@link JtaTransactionInterceptor} in that
39   * case.
40   *
41   * @see JtaTransactionInterceptor
42   *
43   * @author Frank David Mart√≠nez
44   */
45  @Transactional
46  @Interceptor
47  public class LocalTransactionInterceptor implements Serializable {
48  
49    private static final long serialVersionUID = 1L;
50  
51    @Inject
52    private transient SqlSessionManagerRegistry registry;
53  
54    @AroundInvoke
55    public Object invoke(InvocationContext ctx) throws Exception {
56      Transactional transactional = getTransactionalAnnotation(ctx);
57      boolean isInitiator = start(transactional);
58      boolean isExternalJta =  isTransactionActive();
59      if (isInitiator && !isExternalJta) {
60        beginJta();
61      }
62      boolean needsRollback = transactional.rollbackOnly();
63      Object result;
64      try {
65        result = ctx.proceed();
66      } catch (Exception ex) {
67        Exception unwrapped = unwrapException(ex); 
68        needsRollback = needsRollback || needsRollback(transactional, unwrapped);
69        throw unwrapped;
70      } finally {
71        if (isInitiator) {
72          try {
73            if (needsRollback) {
74              rollback(transactional);
75            } else {
76              commit(transactional);
77            }
78          } finally {
79            close();
80            endJta(isExternalJta, needsRollback);
81          }
82        }
83      }
84      return result;
85    }
86  
87    protected boolean isTransactionActive() throws SystemException {
88      return false;
89    }
90      
91    protected void beginJta() throws NotSupportedException, SystemException {
92      // nothing to do
93    }
94  
95    protected void endJta(boolean isExternaTransaction, boolean commit) throws SystemException, RollbackException, HeuristicMixedException, HeuristicRollbackException {
96      // nothing to do
97    }
98  
99    private boolean needsRollback(Transactional transactional, Throwable throwable) {
100     if (RuntimeException.class.isAssignableFrom(throwable.getClass())) {
101       return true;
102     }
103     for (Class<?> exceptionClass : transactional.rollbackFor()) {
104       if (exceptionClass.isAssignableFrom(throwable.getClass())) {
105         return true;
106       }
107     }
108     return false;
109   }
110 
111   protected Transactional getTransactionalAnnotation(InvocationContext ctx) {
112     Transactional t = ctx.getMethod().getAnnotation(Transactional.class);
113     if (t == null) {
114       t = ctx.getMethod().getDeclaringClass().getAnnotation(Transactional.class);
115     }
116     return t;
117   }
118 
119   private boolean start(Transactional transactional) {
120     boolean started = false;
121     for (SqlSessionManager manager : this.registry.getManagers()) {
122       if (!manager.isManagedSessionStarted()) {
123         manager.startManagedSession(transactional.executorType(), transactional.isolation().getTransactionIsolationLevel());
124         started = true;
125       }
126     }
127     return started;
128   }
129 
130   private void commit(Transactional transactional) {
131     for (SqlSessionManager manager : this.registry.getManagers()) {
132       manager.commit(transactional.force());
133     }
134   }
135 
136   private void rollback(Transactional transactional) {
137     for (SqlSessionManager manager : this.registry.getManagers()) {
138       manager.rollback(transactional.force());
139     }
140   }
141 
142   private void close() {
143     for (SqlSessionManager manager : this.registry.getManagers()) {
144       manager.close();
145     }
146   }
147 
148   private Exception unwrapException(Exception wrapped) {
149     Throwable unwrapped = wrapped;
150     while (true) {
151       if (unwrapped instanceof InvocationTargetException) {
152         unwrapped = ((InvocationTargetException) unwrapped).getTargetException();
153       } else if (unwrapped instanceof UndeclaredThrowableException) {
154         unwrapped = ((UndeclaredThrowableException) unwrapped).getUndeclaredThrowable();
155       } else if (!(unwrapped instanceof Exception)) {
156         return new RuntimeException(unwrapped);
157       } else {
158        return (Exception) unwrapped; 
159       }
160     }
161   }
162 
163 }
